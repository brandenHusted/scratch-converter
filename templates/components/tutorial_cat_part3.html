# cat_glhjh313 (pyStage, converted from Scratch 3)

from pystage.en import Sprite, Stage
                
stage = Stage()
stage.add_backdrop('backdrop1')
stage.add_backdrop('desert')
stage.add_backdrop('water_and_rocks')
stage.add_backdrop('blue_sky')
stage.create_variable('your_score')
stage.show_variable("your_score")
stage.set_monitor_position("your_score", -235, 175)
sprite1 = stage.add_a_sprite(None)
sprite1.set_name("Sprite1")
sprite1.set_x(-78)
sprite1.set_y(-109)
sprite1.go_to_back_layer()
sprite1.go_forward(1)
sprite1.add_costume('costume1', center_x=48, center_y=50)
sprite1.add_costume('costume2', center_x=46, center_y=53)
sprite1.next_costume()
sprite1.add_sound('meow')
                    
def when_program_starts_1(self):
    self.think("press a or d to move...")
    self.reset_timer()
    self.set_variable("your_score", 0)
                    
sprite1.when_program_starts(when_program_starts_1)
                    
def when_key_pressed_2(self):
    self.change_x_by(-10.0)
    self.next_costume()
                    
sprite1.when_key_pressed("a", when_key_pressed_2)
                    
def when_key_pressed_3(self):
    self.change_x_by(10.0)
    self.next_costume()
                    
sprite1.when_key_pressed("d", when_key_pressed_3)
                    
def when_key_pressed_4(self):
    if self.touching("NO TRANSLATION: sensing_touchingobjectmenu"):
        self.say("Yum!")
        self.change_variable_by("your_score", 1.0)
                
sprite1.when_key_pressed("d", when_key_pressed_4)
cheesy_puffs = stage.add_a_sprite(None)
cheesy_puffs.set_name("Cheesy Puffs")
cheesy_puffs.set_x(-241)
cheesy_puffs.set_y(-180)
cheesy_puffs.go_to_back_layer()
cheesy_puffs.go_forward(2)
cheesy_puffs.set_size_to(6.944444444444445)
cheesy_puffs.add_costume('cheesy_puffs', center_x=87, center_y=72, factor=2)
cheesy_puffs.add_sound('pop')
                
def when_key_pressed_5(self):
    if self.touching("NO TRANSLATION: sensing_touchingobjectmenu"):
        self.wait(1.0)
        self.set_size_to(0.0)
        self.go_to_x_y(-1000.0, -1000.0)
                
cheesy_puffs.when_key_pressed("d", when_key_pressed_5)
                
def when_program_starts_6(self):
    self.go_to_x_y(150.0, -130.0)
    self.set_size_to(100.0)
                
cheesy_puffs.when_program_starts(when_program_starts_6)
                
stage.play()
